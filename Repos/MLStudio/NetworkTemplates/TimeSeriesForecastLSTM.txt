model = Sequential (
      RecurrentLSTMLayer {$dimension$, usePeepholes = true} : 
      DenseLayer {$labelDimension$, bias=false} 
)

z = model (features)
features=Input($dimension$)
labels=Input($labelDimension$)

# loss and metric
ce = SquareError(labels, z)
errs = SquareError(labels, z)

# node assignment
featureNodes    = (features)
labelNodes      = (labels)
criterionNodes  = (ce)
evaluationNodes = (errs)
outputNodes     = (z)
